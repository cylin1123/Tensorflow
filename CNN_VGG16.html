VGG16¶

VGG16特性¶

  • Convolutional Layer :
    ** conv3-XXX: kernel size = 3 => 3 =>3x3; XXX => channel number, :stride=1padding=same

  • Pool layer:
    ** :22, stride=2, => Pool layer1/2

  • Channel64->128->256->512
  • 224->112->56->28->14->7

import matplotlib.pyplot as plt
from tensorflow.keras.applications.vgg16 import VGG16
from tensorflow.keras.preprocessing import image
from tensorflow.keras.applications.vgg16 import preprocess_input, decode_predictions
import numpy as np
In [3]:
def percent(value):
    return '%.2f%%' % (value * 100)
In [4]:
model = VGG16(weights='imagenet',include_top=True)
In [5]:
img_path = 'data/img/tiger.jpg'
img = image.load_img(img_path, target_size=(224, 224))

plt.imshow(img)
plt.show()
In [6]:
x = image.img_to_array(img) 
x = np.expand_dims(x, axis=0)#tensor size(1, 224, 224, 3)
x = preprocess_input(x)
In [7]:
# features (1,1000)
features = model.predict(x)
In [8]:
# 
pred=decode_predictions(features, top=5)[0]
In [9]:
#
values = []
bar_label = []
for element in pred:
    values.append(element[2])
    bar_label.append(element[1])

fig=plt.figure(u"Top5 Prediction Results ")
ax = fig.add_subplot(111) 
ax.bar(range(len(values)), values, tick_label=bar_label, width=0.5, fc='g')
ax.set_ylabel(u'probability') 
ax.set_title(u'Top5') 
for a,b in zip(range(len(values)), values):
    ax.text(a, b+0.0005, percent(b), ha='center', va = 'bottom', fontsize=9)

fig = plt.gcf()
plt.show()

name=img_path[0:-4]+'_pred'
fig.savefig(name, dpi=300)
